// ── Правильная 3D математика ────────────────────────────────── var worldPos = new THREE.Vector3(0, 0, -3); // Мировая позиция видео var worldQuat = new THREE.Quaternion(); // Мировая ориентация видео var isAnchored = false; function anchorVideo() { // При анкеровке сохраняем текущую мировую позицию var phoneQuat = new THREE.Quaternion().setFromEuler( new THREE.Euler( THREE.MathUtils.degToRad(gyro.pitch), THREE.MathUtils.degToRad(gyro.yaw), 0, 'YXZ' ) ); // Видео в 3 метрах перед камерой в момент анкеровки var forward = new THREE.Vector3(0, 0, -1).applyQuaternion(phoneQuat); worldPos.copy(camera.position.clone().add(forward.multiplyScalar(3))); worldQuat.copy(phoneQuat); isAnchored = true; } function updateVideoPosition() { if (!videoMesh) return; if (isAnchored) { // Видео стоит на месте в мире - вычисляем где оно должно быть // относительно текущей позиции камеры var relativePos = worldPos.clone().sub(camera.position); var distance = relativePos.length(); // Проецируем на плоскость экрана relativePos.applyQuaternion(camera.quaternion.clone().invert()); // Нормализуем для правильной перспективы var scale = 1 / (relativePos.z / distance); // Перспективная коррекция // Обновляем позицию в координатах камеры videoMesh.position.copy(relativePos); videoMesh.quaternion.copy(worldQuat.clone().multiply(camera.quaternion.clone().invert())); } else { // Следует за камерой videoMesh.position.set(0, 0, -3); videoMesh.quaternion.set(0, 0, 0, 1); } }
